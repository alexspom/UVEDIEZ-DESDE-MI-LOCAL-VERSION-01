CREATE USER HDS
  IDENTIFIED BY HDS
  DEFAULT TABLESPACE USERS
  TEMPORARY TABLESPACE TEMP
  PROFILE DEFAULT
  ACCOUNT UNLOCK;
  -- 2 Roles for HDS 
  GRANT CONNECT TO HDS;
  GRANT DBA TO HDS;
  ALTER USER HDS DEFAULT ROLE ALL;
  -- 1 System Privilege for HDS 
  GRANT UNLIMITED TABLESPACE TO HDS;
  -- 2 Object Privileges for HDS 
    GRANT EXECUTE ON MARIO.VDHDS TO HDS;
    GRANT SELECT ON SYS.V_$SESSION TO HDS;
