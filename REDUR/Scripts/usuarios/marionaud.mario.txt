CREATE USER MARIO
  IDENTIFIED BY MARIO
  DEFAULT TABLESPACE USERS
  TEMPORARY TABLESPACE TEMP
  PROFILE DEFAULT
  ACCOUNT UNLOCK;
  -- 2 Roles for MARIO 
  GRANT DBA TO MARIO;
  GRANT CONNECT TO MARIO;
  ALTER USER MARIO DEFAULT ROLE ALL;
  -- 8 System Privileges for MARIO 
  GRANT CREATE ANY INDEX TO MARIO;
  GRANT CREATE ANY TABLE TO MARIO;
  GRANT INSERT ANY TABLE TO MARIO;
  GRANT SELECT ANY TABLE TO MARIO;
  GRANT UPDATE ANY TABLE TO MARIO;
  GRANT CREATE ANY SYNONYM TO MARIO;
  GRANT CREATE ANY TRIGGER TO MARIO;
  GRANT UNLIMITED TABLESPACE TO MARIO;
  -- 6 Object Privileges for MARIO 
    GRANT DELETE, INSERT, SELECT, UPDATE ON HDS.SGML_ORDENES TO MARIO;
    GRANT SELECT ON HDS.SGML_SHUTTLE TO MARIO;
    GRANT SELECT ON HDS.SGML_SILO TO MARIO;
    GRANT EXECUTE ON HDS.SOLICITAR_SALIDA TO MARIO;
    GRANT EXECUTE ON SYS.DBMS_LOCK TO MARIO;
    GRANT SELECT ON SYS.V_$SESSION TO MARIO;
